(defun print-world (world generations)
           (let ((lim (world-size world))
             (current (world-current world)))
         (dotimes (y lim)
           (dotimes (x lim)
             (if (zerop (bit current y x))
                 (princ " ")
                 (princ "*")))
           (terpri))
         (format t "~&~d Generations, ~d Organisms."
                 generations (world-numdots world))))
